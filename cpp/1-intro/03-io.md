# 1.3 输入输出

## cout 输出

```cpp
#include <iostream>
using namespace std;

int main() {
    int age = 25;
    double height = 1.75;
    char grade = 'A';
    string name = "Alice";

    // 基本输出
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;

    // 链式输出
    cout << "Name: " << name << ", Age: " << age << endl;

    return 0;
}
```

## 格式化输出

```cpp
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    double pi = 3.1415926535;
    int num = 42;

    // 字段宽度
    cout << setw(10) << num << endl;

    // 填充字符
    cout << setfill('0') << setw(10) << num << endl;

    // 浮点数精度
    cout << fixed << setprecision(2) << pi << endl;
    cout << fixed << setprecision(5) << pi << endl;

    // 进制
    cout << hex << num << endl;      // 十六进制
    cout << oct << num << endl;      // 八进制
    cout << dec << num << endl;      // 十进制（恢复）

    // 显示进制前缀
    cout << showbase << hex << num << endl;

    // 布尔值
    bool flag = true;
    cout << boolalpha << flag << endl;  // 显示 true/false

    return 0;
}
```

## 文件输入输出

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    // 写入文件
    ofstream outFile("test.txt");
    if (outFile.is_open()) {
        outFile << "Hello, File!" << endl;
        outFile << "Line 2" << endl;
        outFile.close();
    }

    // 读取文件
    ifstream inFile("test.txt");
    if (inFile.is_open()) {
        string line;
        while (getline(inFile, line)) {
            cout << line << endl;
        }
        inFile.close();
    }

    return 0;
}
```

## 字符串流

```cpp
#include <iostream>
#include <sstream>
#include <string>
using namespace std;

int main() {
    // 输出字符串流
    ostringstream oss;
    oss << "Value: " << 42 << ", Pi: " << 3.14;
    string result = oss.str();
    cout << result << endl;

    // 输入字符串流
    istringstream iss("100 200 300");
    int a, b, c;
    iss >> a >> b >> c;
    cout << "Sum: " << (a + b + c) << endl;

    // 解析行
    string line = "Alice 25 95.5";
    istringstream lss(line);
    string name;
    int age;
    double score;
    lss >> name >> age >> score;
    cout << name << " is " << age << " years old" << endl;

    return 0;
}
```

## 错误处理

```cpp
#include <iostream>
using namespace std;

int main() {
    int value;

    cout << "Enter a number: ";

    if (cin >> value) {
        cout << "You entered: " << value << endl;
    } else {
        cout << "Invalid input!" << endl;
        cin.clear();  // 清除错误标志
        cin.ignore(10000, '\n');  // 忽略输入
    }

    return 0;
}
```

## 完整示例

### 简单计算器

```cpp
#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Enter expression (e.g., 3 + 4): ";
    cin >> a >> op >> b;

    cout << fixed << setprecision(2);

    switch (op) {
        case '+':
            cout << "Result: " << (a + b) << endl;
            break;
        case '-':
            cout << "Result: " << (a - b) << endl;
            break;
        case '*':
            cout << "Result: " << (a * b) << endl;
            break;
        case '/':
            if (b != 0) {
                cout << "Result: " << (a / b) << endl;
            } else {
                cout << "Error: Division by zero!" << endl;
            }
            break;
        default:
            cout << "Error: Unknown operator!" << endl;
    }

    return 0;
}
```

### 学生记录

```cpp
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

struct Student {
    string name;
    int id;
    double score;
};

void saveStudent(const Student& s) {
    ofstream file("students.txt", ios::app);
    file << s.name << " " << s.id << " " << s.score << endl;
    file.close();
}

void loadStudents() {
    ifstream file("students.txt");
    Student s;
    while (file >> s.name >> s.id >> s.score) {
        cout << s.name << " (ID: " << s.id
             << ") Score: " << s.score << endl;
    }
    file.close();
}

int main() {
    Student s;

    cout << "Enter student name: ";
    cin >> s.name;
    cout << "Enter ID: ";
    cin >> s.id;
    cout << "Enter score: ";
    cin >> s.score;

    saveStudent(s);

    cout << "\nAll students:" << endl;
    loadStudents();

    return 0;
}
```
